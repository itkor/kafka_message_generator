# Run dockerized kafka
docker-compose up

docker ps

docker exec -it <kafka_conatiner_id> sh

cd /opt/kafka_2.13-2.8.1/bin


/opt/kafka_2.13-2.8.1/bin/kafka-topics.sh --create --zookeeper zookeeper:2181 --replication-factor 1 --partitions 1 --topic test

# Producer
/opt/kafka_2.13-2.8.1/bin/kafka-console-producer.sh --topic test --bootstrap-server localhost:9092

# Consumer
/opt/kafka_2.13-2.8.1/bin/kafka-console-consumer.sh --topic test --from-beginning --bootstrap-server localhost:9092


# Run kafka with automatic topic creation
environment:
      KAFKA_CREATE_TOPICS: "Topic1:1:3,Topic2:1:1:compact"
Topic 1 will have 1 partition and 3 replicas, Topic 2 will have 1 partition, 1 replica and a cleanup.policy set to compact







docker-compose down
